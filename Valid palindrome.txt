class Solution {
public:
    bool isPalindrome(string s) {
        string s1;
        int i,j;
        for(i=0;i<s.length();i++){
            if((s[i]>='a' && s[i]<='z') || (s[i]>='0' && s[i]<='9'))   
                s1+=s[i];
            if(s[i]>='A' && s[i]<='Z')
                s1+=s[i]-'A'+'a';
        }
        i=0,j=s1.length()-1;
        while(i<j){
            if(s1[i]==s1[j]){
                i++;
                j--;
            }
            else
                return false;
        }
        return true;
    }
};
