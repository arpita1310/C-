class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int i,sum=nums[0],n=nums.size();
        int maxele=nums[0];
        if(sum<0)
            sum=0;
        for(i=1;i<n;i++){
            sum+=nums[i];
            if(sum>maxele)
                maxele=sum;
            if(sum<0)
                sum=0;
        }
        return maxele;
    }
};
